function ActResults_Function9

	arrayPos0 <<= 2
	arrayPos0 += 0x400
	temp0 = saveRAM[arrayPos0]
	temp0 &= 15
	temp1 = saveRAM[arrayPos0]
	temp1 >>= 4
	if temp0 == temp1
		temp0 = 0
		temp1 = 0
	else
		if temp0 > temp1
			temp0 = 1
			temp1 = 0
		else
			temp0 = 0
			temp1 = 1
		end if
	end if
	arrayPos0++
	temp2 = saveRAM[arrayPos0]
	temp2 &= 15
	temp3 = saveRAM[arrayPos0]
	temp3 >>= 4
	if temp2 == temp3
		temp2 = 0
		temp3 = 0
	else
		if temp2 > temp3
			temp2 = 1
			temp3 = 0
		else
			temp2 = 0
			temp3 = 1
		end if
	end if
	temp0 += temp2
	temp1 += temp3
	if temp0 == temp1
		checkResult = 1
	else
		checkResult = 0
	end if
end function
