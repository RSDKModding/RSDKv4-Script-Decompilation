event ObjectMain
	switch object.state
	case 0
		object.value0++
		if object.value0 == 64
			object.value0 = 128
			object.xvel = -0x10000
			object.yvel = 0x4000
			object.speed = 512
			object.state = 1
			if object.priority != PRIORITY_XBOUNDS_DESTROY
				object.priority = PRIORITY_ACTIVE
			end if
		end if
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_TOUCH, object.entityPos, -8, -8, 8, 8, currentPlayer, object[currentPlayer].value40, object[currentPlayer].value38, object[currentPlayer].value41, object[currentPlayer].value39)
			if checkResult == 1
				CallFunction(PlayerObject_BadnikBreak)
			end if
		next
		break
	case 1
		if object.value0 > 0
			object.value0--
			if object.xvel < 0
				object.direction = FLIP_NONE
			else
				object.direction = FLIP_X
			end if
			GetTableValue(temp0, object.value2, Flasher_table41)
			object.value1++
			if object.value1 == temp0
				GetTableValue(object.value3, object.value4, Flasher_table42)
				if object.value3 == 1
					FlipSign(object.speed)
				end if
				object.value4++
				GetTableValue(object.value3, object.value4, Flasher_table42)
				if object.value3 == 1
					FlipSign(object.yvel)
				end if
				object.value4++
				object.value4 &= 15
				object.value2++
				if object.value2 == 10
					object.value2 = 0
					object.value1 = 0
				end if
			end if
			object.xvel += object.speed
			object.xpos += object.xvel
			object.ypos += object.yvel
		else
			object.state = 2
		end if
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_TOUCH, object.entityPos, -8, -8, 8, 8, currentPlayer, object[currentPlayer].value40, object[currentPlayer].value38, object[currentPlayer].value41, object[currentPlayer].value39)
			if checkResult == 1
				CallFunction(PlayerObject_BadnikBreak)
			end if
		next
		break
	case 2
		GetTableValue(object.frame, object.value0, Flasher_table43)
		object.value0++
		if object.value0 == 26
			object.value0 = 0
			object.state = 3
		end if
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_TOUCH, object.entityPos, -8, -8, 8, 8, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			if object[currentPlayer].value37 != 4
				if checkResult == 1
					CallFunction(PlayerObject_LightningHit)
				else
					BoxCollisionTest(C_TOUCH, object.entityPos, -8, -8, 8, 8, currentPlayer, object[currentPlayer].value40, object[currentPlayer].value38, object[currentPlayer].value41, object[currentPlayer].value39)
					if checkResult == 1
						CallFunction(PlayerObject_BadnikBreak)
					end if
				end if
			else
				if checkResult == 1
					CallFunction(PlayerObject_BadnikBreak)
				end if
			end if
		next
		break
	case 3
		GetTableValue(object.frame, object.animationTimer, Flasher_table44)
		object.animationTimer++
		if object.animationTimer == 14
			object.animationTimer = 6
		end if
		object.value0++
		if object.value0 == 128
			object.value0 = 0
			object.state = 4
		end if
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			if object[currentPlayer].value37 != 4
				BoxCollisionTest(C_TOUCH, object.entityPos, -12, -12, 12, 12, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
				if checkResult == 1
					CallFunction(PlayerObject_LightningHit)
				else
					BoxCollisionTest(C_TOUCH, object.entityPos, -8, -8, 8, 8, currentPlayer, object[currentPlayer].value40, object[currentPlayer].value38, object[currentPlayer].value41, object[currentPlayer].value39)
					if checkResult == 1
						CallFunction(PlayerObject_BadnikBreak)
					end if
				end if
			else
				BoxCollisionTest(C_TOUCH, object.entityPos, -8, -8, 8, 8, currentPlayer, object[currentPlayer].value40, object[currentPlayer].value38, object[currentPlayer].value41, object[currentPlayer].value39)
				if checkResult == 1
					CallFunction(PlayerObject_BadnikBreak)
				end if
			end if
		next
		break
	case 4
		GetTableValue(object.frame, object.value0, Flasher_table45)
		object.value0++
		if object.value0 == 20
			object.value0 = 128
			object.state = 1
		end if
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_TOUCH, object.entityPos, -8, -8, 8, 8, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			if object[currentPlayer].value37 != 4
				if checkResult == 1
					CallFunction(PlayerObject_LightningHit)
				else
					BoxCollisionTest(C_TOUCH, object.entityPos, -8, -8, 8, 8, currentPlayer, object[currentPlayer].value40, object[currentPlayer].value38, object[currentPlayer].value41, object[currentPlayer].value39)
					if checkResult == 1
						CallFunction(PlayerObject_BadnikBreak)
					end if
				end if
			else
				if checkResult == 1
					CallFunction(PlayerObject_BadnikBreak)
				end if
			end if
		next
		break
	end switch
	if object.outOfBounds == 1
		temp0 = object.xpos
		temp1 = object.ypos
		object.xpos = object.value5
		object.ypos = object.value6
		if object.outOfBounds == 1
			if object.priority != PRIORITY_XBOUNDS_DESTROY
				object.value0 = 0
				object.frame = 0
				object.state = 0
				object.priority = PRIORITY_ACTIVE_BOUNDS
			end if
		else
			object.xpos = temp0
			object.ypos = temp1
		end if
	end if
end event
