	DrawSprite(0)

function Bubbler_DebugSpawn
	CreateTempObject(TypeName[Bubbler], 0, object.xpos, object.ypos)
end function


event ObjectMain
	object.priority = PRIORITY_ACTIVE
	switch object.state
	case 0
		object.animationTimer++
		object.animationTimer &= 15
		object.frame = object.animationTimer
		object.frame >>= 2
		object.ypos += 0x10000
		ObjectTileCollision(CSIDE_FLOOR, 0, 0, 0)
		if checkResult == 1
			object.state = 1
			object.animationTimer = 0
			object.frame = 4
		end if
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			BoxCollisionTest(C_TOUCH, object.entityPos, -2, -2, 2, 2, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			if checkResult == 1
				CallFunction(PlayerObject_ProjectileHit)
			end if
		next
		break
	case 1
		object.animationTimer++
		if object.animationTimer >= 4
			object.animationTimer = 0
			object.frame++
			if object.frame >= 11
				object.type = TypeName[Blank Object]
			end if
		end if
		if object.frame < 10
			temp0 = object.frame
			temp0 -= 4
			temp0 >>= 1
			temp0 <<= 2
			temp0 += 4
			temp1 = temp0
			FlipSign(temp0)
			foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
				BoxCollisionTest(C_TOUCH, object.entityPos, temp0, -8, temp1, 0, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
				if checkResult == 1
					CallFunction(PlayerObject_Hit)
				end if
			next
		end if
		break
	end switch
end event


event ObjectDraw
	DrawSprite(object.frame)
end event
