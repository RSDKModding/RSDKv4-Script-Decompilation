event ObjectMain
	object.xpos += object.xvel
	object.ypos += object.yvel
	object.animationTimer++
	if object.animationTimer >= 3
		object.animationTimer = 0
		object.frame ^= 1
	end if
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		BoxCollisionTest(C_TOUCH, object.entityPos, -4, -4, 4, 4, currentPlayer, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
		if checkResult == 1
			CallFunction(PlayerObject_ProjectileHit)
		end if
	next
	temp0 = object.xpos
	temp0 >>= 16
	temp0 -= screen.cameraX
	Abs(temp0)
	temp0 -= 32
	temp1 = object.ypos
	temp1 >>= 16
	temp1 -= screen.cameraY
	Abs(temp1)
	temp1 -= 32
	if temp0 >= screen.xcenter
		object.type = TypeName[Blank Object]
	end if
	if temp1 >= screen.ycenter
		object.type = TypeName[Blank Object]
	end if
end event


event ObjectDraw
	DrawSprite(object.frame)
end event

	else

		SpriteFrame(-4, -4, 8, 8, 973, 677)
		SpriteFrame(-4, -4, 8, 8, 982, 677)
	end if
end event
