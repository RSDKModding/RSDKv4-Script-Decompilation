// Static Values

// Tables

function Turret_DebugDraw
	DrawSprite(0)
end function


function Turret_DebugSpawn
	CreateTempObject(TypeName[Turret], 0, object.xpos, object.ypos)
end function


event ObjectMain
	if object.state == 0
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			if object[currentPlayer].ypos >= object.ypos
				temp0 = object[currentPlayer].xpos
				temp0 -= object.xpos
				Abs(temp0)
				if temp0 < 0x600000
					temp0 = object[currentPlayer].ypos
					temp0 -= object.ypos
					if temp0 < 0x900000
						object.value0 = 2
						object.state++
					end if
				end if
			end if
		next
	else
		temp1 = 0x7FFFFFFF
		foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
			temp0 = object[currentPlayer].xpos
			temp0 -= object.xpos
			Abs(temp0)
			if temp0 < temp1
				if temp0 >= 0x200000
					if object[currentPlayer].xpos < object.xpos
						object.frame = 1
					else
						object.frame = 2
					end if
				else
					object.frame = 0
				end if
			end if
		next
		object.value0--
		if object.value0 == 0
			object.value0 = 96
			PlaySfx(SfxName[Octus Shot], 0)
			CreateTempObject(TypeName[Turret Bullet], 0, object.xpos, object.ypos)
			switch object.frame
			case 0
				object[tempObjectPos].ypos += 0x180000
				object[tempObjectPos].yvel = 0x10000
				break
			case 1
				object[tempObjectPos].xpos -= 0x110000
				object[tempObjectPos].ypos += 0x100000
				object[tempObjectPos].xvel = -0x10000
				object[tempObjectPos].yvel = 0x10000
				break
			case 2
				object[tempObjectPos].xpos += 0x110000
				object[tempObjectPos].ypos += 0x100000
				object[tempObjectPos].xvel = 0x10000
				object[tempObjectPos].yvel = 0x10000
				break
			end switch
		end if
	end if
end event


event ObjectDraw
	DrawSprite(object.frame)
end event
