	temp1 -= 0x400000

		FlipSign(temp0)
		FlipSign(temp1)
	end if
	GetBit(temp2, object.propertyValue, 6)

		FlipSign(temp0)
	end if

		object.value2 = temp0
		object.value3 = temp1

		FlipSign(temp0)

		object.value3 = temp0

	object.value3 += object.value1
	temp6 = object.value2
	temp6 -= temp4
	temp7 = object.value3
	temp7 -= temp5
