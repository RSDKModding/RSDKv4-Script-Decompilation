					if checkResult == 4
						if object.propertyValue == 0
							CallFunction(PlayerObject_Kill)
						end if
					end if
				end if
			next
			object.xpos += temp4
			temp2++
		loop
		if object.propertyValue == 0
			object.value1 += 0x4000
		else
			object.value1 -= 0x4000
		end if
		object.xpos = temp0
		object.ypos = temp1
		object.value0--

			temp2 += 0x600000
			temp4 = -0x200000
		else
			temp4 = 0x200000
		end if
