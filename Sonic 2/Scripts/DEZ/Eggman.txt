event ObjectMain
	switch object.state
	case 0
		object[-1].state = 0
		BoxCollisionTest(C_TOUCH, object.entityPos, -92, -92, 0, 92, 0, 1, 1, 1, 1)
		if checkResult == 1
			object.frame = 1
			object.state = 1
		end if
		break
	case 1
		object[-1].state = 0
		object.animationTimer++
		if object.animationTimer >= 24
			object.state = 2
		end if
		break
	case 2
		if object[0].xpos < object[+1].xpos
			object[+1].state = 1
		end if
		object.xpos += 0x20000
		temp0 = object.xpos
		temp0 -= 0x500000
		if object[0].xpos > temp0
			object.xpos = object[0].xpos
			object.xpos += 0x500000
		end if
		object.animationTimer++
		if object.animationTimer >= 6
			object.animationTimer = 0
			object.frame++
			if object.frame > 4
				object.frame = 2
			end if
		end if
		object.value4++
		switch object.value4
		case 1
			object.value0 = object.xpos
			object.value0 -= 0x70000
			object.value1 = object.ypos
			object.value1 -= 0x180000
			object.value2 = 0
			object.value3 = 1
			break
		case 2
		case 3
		case 4
		case 5
		case 6
		case 7
		case 8
			object.value0 -= 0x10000
			object.value2 += 0x1C00
			object.value1 += object.value2
			break
		case 9
			object.value3 = 0
			break
		case 32
			object.value4 = 0
			break
		end switch
		temp0 = object[+9].xpos
		temp0 -= object.xpos
		if temp0 <= 0x2C0000
			object.xpos = object[+9].xpos
			object.xpos -= 0x2C0000
			object.yvel = -0x20000
			object.frame = 2
			object.value3 = 0
			object.state = 3
		end if
		break
	case 3
		object.xpos += 0xA000
		object.ypos += object.yvel
		object.yvel += 0x1000
		if object.ypos >= object[+9].ypos
			PlaySfx(SfxName[Machine Move], 0)
			object[+9].state = 2
			object.type = TypeName[Blank Object]
		end if
		break
	end switch
end event

	DrawSprite(object.frame)
end event
