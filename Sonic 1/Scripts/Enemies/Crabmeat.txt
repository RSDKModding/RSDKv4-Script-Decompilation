private alias 42 : TYPE_CRABMEAT

reserve function Crabmeat_DebugDraw

// Static Values

function Crabmeat_DebugDraw
	DrawSprite(0)
end function


function Crabmeat_DebugSpawn
	CreateTempObject(TypeName[Crabmeat], 0, object.xpos, object.ypos)
	object[tempObjectPos].value1 = object.xpos
	object[tempObjectPos].value2 = object.ypos
	GetBit(temp0, object.direction, 0)
	if temp0 == 0
		object[tempObjectPos].direction = FLIP_X
		object[tempObjectPos].xvel = 0x8000
	else
		object[tempObjectPos].direction = FLIP_X
		object[tempObjectPos].xvel = -0x8000
	end if
	object[tempObjectPos].value3 = 0
end function


event ObjectMain
	switch object.state
	case 0
		object.priority = PRIORITY_ACTIVE
		if object.value0 < 127
			object.value0++
			object.xpos += object.xvel
			if object.xvel > 0
				ObjectTileGrip(CSIDE_FLOOR, 14, 15, 0)
			else
				ObjectTileGrip(CSIDE_FLOOR, -14, 15, 0)
			end if
			if checkResult == 0
				if object.value3 == 1
					object.state = 2
				else
					object.state = 1
					object.value3 = 1
				end if
				object.value0 = 0
				object.animation = 1
			end if
			ObjectTileGrip(CSIDE_FLOOR, 0, 15, 0)
		else
			if object.value3 == 1
				object.state = 2
			else
				object.state = 1
				object.value3 = 1
			end if
			object.value0 = 0
			object.animation = 1
		end if
		break
	case 1
		if object.value0 < 59
			object.value0++
		else
			object.value0 = 0
			object.state = 0
			object.animation = 0
			object.animationTimer = 0
			FlipSign(object.xvel)
		end if
		break
	case 2
		if object.value0 < 59
			object.value0++
		else
			CreateTempObject(TypeName[Crabmeat Shot], 0, object.xpos, object.ypos)
			object[tempObjectPos].xpos -= 0x100000
			object[tempObjectPos].xvel = -0x10000
			object[tempObjectPos].yvel = -0x40000
			CreateTempObject(TypeName[Crabmeat Shot], 0, object.xpos, object.ypos)
			object[tempObjectPos].xpos += 0x100000
			object[tempObjectPos].xvel = 0x10000
			object[tempObjectPos].yvel = -0x40000
			object.value0 = 0
			object.state++
			object.animation = 2
		end if
		break
	case 3
		if object.value0 < 59
			object.value0++
		else
			object.value0 = 0
			object.state = 0
			object.animation = 0
			object.animationTimer = 0
			FlipSign(object.xvel)
		end if
		break
	end switch
	if object.outOfBounds == 1
		temp0 = object.xpos
		temp1 = object.ypos
		object.xpos = object.value1
		object.ypos = object.value2
		if object.outOfBounds == 1
			if object.propertyValue == 1
				object.direction = FLIP_X
				object.xvel = 0x8000
			else
				object.direction = FLIP_X
				object.xvel = -0x8000
			end if
			object.value3 = 0
			object.state = 0
			object.value0 = 0
			object.priority = PRIORITY_ACTIVE_BOUNDS
			object.animation = 0
			object.animationTimer = 0
		else
			object.xpos = temp0
			object.ypos = temp1
		end if
	end if
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		BoxCollisionTest(C_TOUCH, object.entityPos, -14, -14, 14, 14, currentPlayer, object[currentPlayer].value40, object[currentPlayer].value38, object[currentPlayer].value41, object[currentPlayer].value39)
		if checkResult == 1
			CallFunction(PlayerObject_BadnikBreak)
		end if
	next
	switch object.animation
	case 0
		object.frame = object.animationTimer
		object.frame >>= 4
		object.animationTimer++
		object.animationTimer %= 48
		break
	case 1
		object.frame = 0
		break
	case 2
		object.frame = 3
		break
	end switch
end event


event ObjectDraw
	DrawSprite(object.frame)
end event
