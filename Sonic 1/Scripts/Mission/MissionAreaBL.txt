event ObjectUpdate
	if object.state == 0
		stage.newXBoundary1 = object.xpos
		stage.newXBoundary1 >>= 16
		stage.newXBoundary1 -= 16
		stage.newYBoundary2 = object.ypos
		stage.newYBoundary2 >>= 16
		stage.newYBoundary2 += 16
		object.state = 1
	end if
	
	if stage.state == STAGE_RUNNING
		if object[0].ypos > object.ypos
			object[0].yvel = 0
			object[0].speed = 0
			object[0].ypos = object.ypos
			game.forceKillPlayer = true
			object.priority = 5
		end if
		
		if object[0].xpos < object.xpos
			object[0].xvel = 0
			object[0].speed = 0
			object[0].xpos = object.xpos
		end if
	end if
end event
